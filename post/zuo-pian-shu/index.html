左偏树

2020-06-29

#include <bits/stdc++.h>
using namespace std;
const int maxn = 100010;
int n, m, op, x, y;
int lson[maxn], rson[maxn], dist[maxn], fa[maxn];
bool tf[maxn];
struct node{
int id, v;
bool operator<(node x) const {
return v == x.v ? id < x.id : v < x.v;
}
} v[maxn];
int find(int x){
return fa[x] == x ? x : fa[x] = find(fa[x]);
}
int merge(int x, int y){
if (!x || !y)
return x + y;
if (v[y] < v[x])
swap(x, y);
rson[x] = merge(rson[x], y);
if (dist[lson[x]] < dist[rson[x]])
swap(lson[x], rson[x]);
dist[x] = dist[rson[x]] + 1;
return x;
}
int main()
{
dist[0] = -1;
read(n);
read(m);
For(i,1,n)
read(v[i].v), fa[i] = i, v[i].id = i;
while (m--)
{
read(op);
read(x);
if (op == 1)
{
read(y);
if (tf[x] || tf[y])
continue;
x = find(x);
y = find(y);
if (x != y)
fa[x] = fa[y] = merge(x, y);
}
if (op == 2)
{
if (tf[x])
{
printf("-1\n");
continue;
}
x = find(x);
printf("%d\n", v[x].v);
tf[x] = true;
fa[lson[x]] = fa[rson[x]] = fa[x] = merge(lson[x], rson[x]);
lson[x] = rson[x] = dist[x] = 0;
}
}
return 0;
}
